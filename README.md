### setup bash aliases

    cat << EOF > /tmp/bash_aliases
    # ls aliases
    alias ll='ls -l'
    alias la='ls -al'

    # navigation aliases
    alias ..='cd ..'

    # docker aliases
    EOF
    mv /tmp/bash_aliases ~/.bash_aliases
